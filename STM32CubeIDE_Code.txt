/* ================= MORPHED OUTPUT PARAMETERS ================= */
/*
 * Set INPUT_VOLTAGE to your actual inverter input voltage (measured).
 * All V/f scaling is derived from this — do not use mains 230V here.
 *
 * True constant V/f law:  V = VF_BOOST + VF_SLOPE x F
 *   VF_BOOST : small fixed voltage at low speed to overcome stator
 *              resistance (~5-6% of input voltage, here ~2V for 35V input)
 *   VF_SLOPE : (INPUT_VOLTAGE - VF_BOOST) / BASE_FREQ  = 0.66 V/Hz
 *
 * Resulting V/f ratio stays ~0.70 V/Hz across all speeds:
 *   RPM  300 → F=10.0Hz  V= 8.6V   V/f=0.86
 *   RPM  500 → F=16.7Hz  V=13.0V   V/f=0.78
 *   RPM  800 → F=26.7Hz  V=19.6V   V/f=0.73   (your ~22V target ✓)
 *   RPM 1000 → F=33.3Hz  V=24.0V   V/f=0.72
 *   RPM 1500 → F=50.0Hz  V=35.0V   V/f=0.70   ← rated
 */
#define INPUT_VOLTAGE     35.0f      // Actual inverter input voltage (V rms) – TUNE THIS
#define BASE_CURRENT      5.0f       // Rated current at base frequency (A rms) – TUNE THIS

#define VF_BOOST          2.0f                                    // Low-speed boost (V), ~5% of INPUT_VOLTAGE
#define VF_SLOPE          ((INPUT_VOLTAGE - VF_BOOST) / BASE_FREQ)  // V/Hz = 0.66 for 35V input

/* Lightweight 16-bit Galois LFSR – no stdlib, ~5 cycles, deterministic */
static uint16_t lfsr_state = 0xACE1u;

static float lfsr_noise(void)
{
    uint16_t lsb = lfsr_state & 1u;
    lfsr_state >>= 1;
    if (lsb) lfsr_state ^= 0xB400u;
    return ((float)lfsr_state / 32767.5f) - 1.0f;   /* -1.0 … +1.0 */
}

/* ================= UART ================= */
void usart2_send_char(char c)
{
    while (!(USART2->SR & USART_SR_TXE));
    USART2->DR = c;
}

void usart2_print(const char *s)
{
    while (*s) usart2_send_char(*s++);
}

void usart2_print_float(float v)
{
    char buf[32];
    snprintf(buf, sizeof(buf), "%.2f", v);
    usart2_print(buf);
}

/* ================= CLOCK ================= */
void clock_init(void)
{
    RCC->CR |= RCC_CR_HSION;
    while (!(RCC->CR & RCC_CR_HSIRDY));

    RCC->PLLCFGR =
        (16 << RCC_PLLCFGR_PLLM_Pos) |
        (336 << RCC_PLLCFGR_PLLN_Pos) |
        (1 << RCC_PLLCFGR_PLLP_Pos) |
        (7 << RCC_PLLCFGR_PLLQ_Pos);

    RCC->CR |= RCC_CR_PLLON;
    while (!(RCC->CR & RCC_CR_PLLRDY));

    FLASH->ACR = FLASH_ACR_LATENCY_2WS;
    RCC->CFGR |= RCC_CFGR_SW_PLL;
    while ((RCC->CFGR & RCC_CFGR_SWS) != RCC_CFGR_SWS_PLL);
}

/* ================= GPIO ================= */
void gpio_init(void)
{
    RCC->AHB1ENR |= RCC_AHB1ENR_GPIOAEN | RCC_AHB1ENR_GPIOBEN;

    // PA0 - Analog (voltage sensor)
    GPIOA->MODER |= (3 << 0);

    // PA1 - Analog (current sensor ACS712)
    GPIOA->MODER |= (3 << 2);

    // PA4 (sensor pin) - Input with pull-down
    GPIOA->MODER &= ~(3 << (SENSOR_PIN * 2));
    GPIOA->PUPDR |= (2 << (SENSOR_PIN * 2));

    // PA5 - Output (emergency stop indicator)
    GPIOA->MODER |= (1 << 10);
    GPIOA->BSRR = (1 << (5 + 16));

    // PA8, PA9, PA10 - Alternate function (TIM1)
    GPIOA->MODER |= (2<<(8*2)) | (2<<(9*2)) | (2<<(10*2));
    GPIOA->AFR[1] |= (1<<(0*4)) | (1<<(1*4)) | (1<<(2*4));

    // PB13, PB0, PB1 - Alternate function (TIM1)
    GPIOB->MODER |= (2<<(13*2)) | (2<<(0*2)) | (2<<(1*2));
    GPIOB->AFR[1] |= (1<<(5*4));
    GPIOB->AFR[0] |= (1<<(0*4)) | (1<<(1*4));
}

/* ================= USART2 ================= */
void usart2_init(void)
{
    RCC->APB1ENR |= RCC_APB1ENR_USART2EN;

    GPIOA->MODER |= (2 << (2 * 2)) | (2 << (3 * 2));
    GPIOA->AFR[0] |= (7 << (2 * 4)) | (7 << (3 * 4));

    USART2->BRR = 42000000 / 115200;

    USART2->CR1 =
        USART_CR1_TE |
        USART_CR1_RE |
        USART_CR1_RXNEIE |
        USART_CR1_UE;

    NVIC_EnableIRQ(USART2_IRQn);
}

/* ================= UART RX ISR ================= */
void USART2_IRQHandler(void)
{
    if (USART2->SR & USART_SR_RXNE)
    {
        char c = USART2->DR;

        if (c == '\r' || c == '\n')
        {
            rx_buffer[rx_index] = '\0';
            if (rx_index > 0) new_command = 1;
            rx_index = 0;
        }
        else if (rx_index < RX_BUF_SIZE - 1)
        {
            rx_buffer[rx_index++] = c;
        }
    }
}

/* ================= COMMAND PROCESSING ================= */
void process_command(void)
{
    char *cmd = (char*)rx_buffer;

    if (!strncmp(cmd, "S:", 2))
    {
        int new_rpm = atoi(&cmd[2]);
        if (new_rpm >= 0 && new_rpm <= 1500)
        {
            rpm_reference = (float)new_rpm;
            target_frequency = (POLES * rpm_reference) / 120.0f;
            if (target_frequency < 2.0f) target_frequency = 2.0f;
            if (target_frequency > BASE_FREQ) target_frequency = BASE_FREQ;

            ctrl_state = STARTUP_OPEN_LOOP;
            pi_integral = 0;
            stable_time = 0;
            overshoot_count = 0;
            max_overshoot = 0;

            usart2_print("ACK SPEED ");
            usart2_print_float(rpm_reference);
            usart2_print("\r\n");
        }
        else
        {
            usart2_print("ERR: RPM range 0-1500\r\n");
        }
    }
    else if (!strcmp(cmd, "KILL"))
    {
        eStop = 1;
        GPIOA->BSRR = (1 << 5);
        usart2_print("EMERGENCY STOP\r\n");
    }
    else if (!strcmp(cmd, "RESET"))
    {
        eStop = 0;
        GPIOA->BSRR = (1 << (5 + 16));
        ctrl_state = STARTUP_OPEN_LOOP;
        pi_integral = 0;
        stable_time = 0;
        usart2_print("RESET COMPLETE\r\n");
    }
    else if (!strcmp(cmd, "VCAL"))
    {
        char buf[256];
        snprintf(buf, sizeof(buf),
                 "=== VOLTAGE SENSOR ===\r\n"
                 "DC Offset: %.1f\r\n"
                 "Peak+: %.2fV, Peak-: %.2fV\r\n"
                 "RMS: %.2fV\r\n"
                 "Cal Factor: %.4f\r\n",
                 dc_offset,
                 voltage_peak_positive,
                 voltage_peak_negative,
                 voltage_rms_filtered,
                 calibration_factor);
        usart2_print(buf);
    }
    else if (!strcmp(cmd, "ICAL"))
    {
        char buf[256];
        snprintf(buf, sizeof(buf),
                 "=== CURRENT SENSOR ===\r\n"
                 "DC Offset: %.1f\r\n"
                 "Peak+: %.2fA, Peak-: %.2fA\r\n"
                 "RMS: %.3fA\r\n"
                 "Calibrated: %s\r\n",
                 current_dc_offset,
                 current_peak_positive,
                 current_peak_negative,
                 current_rms_filtered,
                 current_offset_calibrated ? "YES" : "NO");
        usart2_print(buf);
    }
    else if (!strcmp(cmd, "STATS"))
    {
        char buf[256];
        snprintf(buf, sizeof(buf),
                 "=== STATISTICS ===\r\n"
                 "Overshoot: %lu (Max: %.1f RPM)\r\n"
                 "Total Pulses: %lu\r\n"
                 "RPM Valid: %d\r\n"
                 "Power: %.2fW\r\n",
                 overshoot_count, max_overshoot,
                 total_pulses,
                 rpm_valid,
                 voltage_rms_filtered * current_rms_filtered);
        usart2_print(buf);
    }
    else
    {
        usart2_print("Commands: S:rpm, KILL, RESET, VCAL, ICAL, STATS\r\n");
    }
}

/* ================= SINE LUT ================= */
void generate_sine_lut(void)
{
    for (int i = 0; i < LUT_SIZE; i++)
    {
        float s = 0.5f * (sinf(2.0f * PI_CONST * i / LUT_SIZE) + 1.0f);
        sine_lut[i] = (uint16_t)(s * 1000.0f);
    }
}

/* ================= TIM1 PWM INIT ================= */
void TIM1_PWM_Init(void)
{
    RCC->APB2ENR |= RCC_APB2ENR_TIM1EN;
a
    TIM1->PSC = 0;
    TIM1->ARR = (SYSCLK / PWM_FREQ) - 1;

    TIM1->CCMR1 =
        (6 << TIM_CCMR1_OC1M_Pos) | TIM_CCMR1_OC1PE |
        (6 << TIM_CCMR1_OC2M_Pos) | TIM_CCMR1_OC2PE;

    TIM1->CCMR2 =
        (6 << TIM_CCMR2_OC3M_Pos) | TIM_CCMR2_OC3PE;

    TIM1->CCER =
        TIM_CCER_CC1E | TIM_CCER_CC1NE |
        TIM_CCER_CC2E | TIM_CCER_CC2NE |
        TIM_CCER_CC3E | TIM_CCER_CC3NE;

    TIM1->BDTR = DEAD_TIME | TIM_BDTR_MOE;
    TIM1->DIER |= TIM_DIER_UIE;
    TIM1->CR1 |= TIM_CR1_ARPE;

    NVIC_EnableIRQ(TIM1_UP_TIM10_IRQn);
    TIM1->CR1 |= TIM_CR1_CEN;
}

/* ================= TIM1 ISR ================= */
void TIM1_UP_TIM10_IRQHandler(void)
{
    if (TIM1->SR & TIM_SR_UIF)
    {
        TIM1->SR &= ~TIM_SR_UIF;

        if (eStop)
        {
            TIM1->CCR1 = 0;
            TIM1->CCR2 = 0;
            TIM1->CCR3 = 0;
            return;
        }

        float step = (frequency * LUT_SIZE) / PWM_FREQ;
        phase_acc += step;
        if (phase_acc >= LUT_SIZE) phase_acc -= LUT_SIZE;

        uint16_t a = (uint16_t)phase_acc;
        uint16_t b = (a + 2 * LUT_SIZE / 3) % LUT_SIZE;
        uint16_t c = (a + LUT_SIZE / 3) % LUT_SIZE;

        uint16_t arr = TIM1->ARR;
        TIM1->CCR1 = (sine_lut[a] * arr * v_scale) / 1000;
        TIM1->CCR2 = (sine_lut[b] * arr * v_scale) / 1000;
        TIM1->CCR3 = (sine_lut[c] * arr * v_scale) / 1000;
    }
}

/* ================= ADC INIT ================= */
void ADC1_Init(void)
{
    RCC->APB2ENR |= RCC_APB2ENR_ADC1EN;

    ADC1->CR2 = ADC_CR2_ADON;
    for (volatile int i = 0; i < 10000; i++);

    ADC1->SMPR2 |= (7 << 0);  // Channel 0 (PA0 - voltage)
    ADC1->SMPR2 |= (7 << 3);  // Channel 1 (PA1 - current)

    ADC1->SQR3 = 0;
    ADC1->SQR1 = 0;

    ADC1->CR2 |= ADC_CR2_EXTEN_0;
    ADC1->CR2 |= (6 << ADC_CR2_EXTSEL_Pos);

    ADC1->CR1 |= ADC_CR1_EOCIE;
    NVIC_EnableIRQ(ADC_IRQn);

    ADC1->CR2 |= ADC_CR2_ADON;
}

/* ================= VOLTAGE CALCULATION ================= */
void Calculate_Voltage_RMS(void)
{
    if (!dc_offset_calibrated)
    {
        dc_offset_samples++;

        if (dc_offset_samples < 2560)
            dc_offset = 0.95f * dc_offset + 0.05f * (float)adc_buffer[adc_index];
        else if (dc_offset_samples < 64000)
            dc_offset = 0.999f * dc_offset + 0.001f * (float)adc_buffer[adc_index];
        else
        {
            dc_offset_calibrated = 1;
            char buf[64];
            snprintf(buf, sizeof(buf), ">>> VOLTAGE DC OFFSET: %.1f\r\n", dc_offset);
            usart2_print(buf);
        }
        return;
    }

    dc_offset = 0.9999f * dc_offset + 0.0001f * (float)adc_buffer[adc_index];

    float adc_centered = (float)adc_buffer[adc_index] - dc_offset;
    float v_instantaneous = adc_centered * ADC_SCALE;

    if (v_instantaneous > voltage_peak_positive)
        voltage_peak_positive = v_instantaneous;
    if (v_instantaneous < voltage_peak_negative)
        voltage_peak_negative = v_instantaneous;

    voltage_squared_sum += v_instantaneous * v_instantaneous;
    rms_sample_count++;

    if (rms_sample_count >= ADC_SAMPLES_PER_CYCLE)
    {
        voltage_rms = sqrtf(voltage_squared_sum / (float)ADC_SAMPLES_PER_CYCLE);

        #if AUTO_CALIBRATE
        if (voltage_rms > 0.01f)
        {
            float new_factor = KNOWN_INPUT_VOLTAGE / voltage_rms;
            calibration_factor = 0.99f * calibration_factor + 0.01f * new_factor;
        }
        #endif

        float voltage_calibrated = voltage_rms * calibration_factor;
        voltage_rms_filtered = 0.9f * voltage_rms_filtered + 0.1f * voltage_calibrated;

        voltage_squared_sum = 0;
        rms_sample_count = 0;
    }
}

/* ================= CURRENT CALCULATION (ACS712) ================= */
void Calculate_Current_RMS(void)
{
    if (!current_offset_calibrated)
    {
        current_offset_samples++;

        if (current_offset_samples < 2560)
            current_dc_offset = 0.95f * current_dc_offset + 0.05f * (float)current_adc_buffer[current_adc_index];
        else if (current_offset_samples < 64000)
            current_dc_offset = 0.999f * current_dc_offset + 0.001f * (float)current_adc_buffer[current_adc_index];
        else
        {
            current_offset_calibrated = 1;
            char buf[64];
            snprintf(buf, sizeof(buf), ">>> CURRENT DC OFFSET: %.1f\r\n", current_dc_offset);
            usart2_print(buf);
        }
        return;
    }

    current_dc_offset = 0.9999f * current_dc_offset + 0.0001f * (float)current_adc_buffer[current_adc_index];

    float adc_centered = (float)current_adc_buffer[current_adc_index] - current_dc_offset;
    float v_sensor = adc_centered * ADC_SCALE;
    float current_instantaneous = v_sensor / ACS712_SENSITIVITY;

    if (current_instantaneous > current_peak_positive)
        current_peak_positive = current_instantaneous;
    if (current_instantaneous < current_peak_negative)
        current_peak_negative = current_instantaneous;

    current_squared_sum += current_instantaneous * current_instantaneous;
    current_sample_count++;

    if (current_sample_count >= CURRENT_SAMPLES)
    {
        current_rms = sqrtf(current_squared_sum / (float)CURRENT_SAMPLES);
        current_rms_filtered = 0.9f * current_rms_filtered + 0.1f * current_rms;
        current_squared_sum = 0;
        current_sample_count = 0;
    }
}

/* ================= ADC ISR ================= */
void ADC_IRQHandler(void)
{
    if (ADC1->SR & ADC_SR_EOC)
    {
        uint16_t adc_value = ADC1->DR;

        if (current_adc_channel == 0)
        {
            adc_buffer[adc_index] = adc_value;
            Calculate_Voltage_RMS();

            adc_index++;
            if (adc_index >= ADC_SAMPLES_PER_CYCLE) adc_index = 0;

            ADC1->SQR3 = CURRENT_ADC_CHANNEL;
            current_adc_channel = 1;
        }
        else
        {
            current_adc_buffer[current_adc_index] = adc_value;
            Calculate_Current_RMS();

            current_adc_index++;
            if (current_adc_index >= CURRENT_SAMPLES) current_adc_index = 0;

            ADC1->SQR3 = 0;
            current_adc_channel = 0;
        }
    }
}

/* ================= RPM MEASUREMENT ================= */
void update_rpm(void)
{
    uint32_t now = msTicks;

    uint8_t sensor_state = (GPIOA->IDR >> SENSOR_PIN) & 1;

    if (sensor_state && !last_sensor_state)
    {
        pulseCount++;
        total_pulses++;
        lastPulseTime = now;
        rpm_valid = 1;
    }

    last_sensor_state = sensor_state;

    if (rpm_valid && (now - lastPulseTime) > RPM_TIMEOUT_MS)
    {
        rpm_avg = 0;
        rpm_ema = 0;
        rpm_valid = 0;
        for (int i = 0; i < AVG_SAMPLES; i++) rpm_buf[i] = 0;
        usart2_print(">>> RPM TIMEOUT - Motor stopped?\r\n");
    }

    if ((now - lastRpmTick) >= SAMPLE_TIME_MS)
    {
        /* RPM = (pulses / pulses_per_rev) * (60000 ms/min / sample_ms) */
        float rpm_inst = ((float)pulseCount / (float)PULSES_PER_REV)
                         * (60000.0f / (float)SAMPLE_TIME_MS);

        rpm_buf[rpm_idx++] = rpm_inst;
        if (rpm_idx >= AVG_SAMPLES) rpm_idx = 0;

        float sum = 0;
        for (int i = 0; i < AVG_SAMPLES; i++) sum += rpm_buf[i];
        rpm_avg = sum / AVG_SAMPLES;

        if (rpm_ema == 0) rpm_ema = rpm_inst;
        else rpm_ema = RPM_EMA_ALPHA * rpm_inst + (1.0f - RPM_EMA_ALPHA) * rpm_ema;

        rpm_avg = rpm_ema;

        pulseCount = 0;
        lastRpmTick = now;
    }
}

/* ================= PI CONTROLLER ================= */
void pi_controller(void)
{
    float error_raw = rpm_reference - rpm_avg;
    speed_error_filtered += ERROR_FILTER_ALPHA * (error_raw - speed_error_filtered);
    speed_error = speed_error_filtered;

    if (rpm_avg > rpm_reference)
    {
        float overshoot = rpm_avg - rpm_reference;
        if (overshoot > max_overshoot) max_overshoot = overshoot;
        if (overshoot > RPM_DEADBAND) overshoot_count++;
    }

    float error_magnitude = fabsf(speed_error);
    if (error_magnitude < RPM_DEADBAND) speed_error = 0;

    switch (ctrl_state)
    {
        case STARTUP_OPEN_LOOP:
        {
            float tolerance = TRANSITION_TOLERANCE * rpm_reference;
            if (rpm_reference < 10.0f) break;
            if (error_magnitude < tolerance && rpm_avg <= rpm_reference * 1.05f)
            {
                ctrl_state = STABILIZING;
                stable_time = msTicks;
                usart2_print(">>> STABILIZING\r\n");
            }
            break;
        }

        case STABILIZING:
        {
            float tolerance = TRANSITION_TOLERANCE * rpm_reference;
            if (rpm_avg > rpm_reference * 1.1f)
            {
                ctrl_state = STARTUP_OPEN_LOOP;
                pi_integral = 0;
                usart2_print(">>> OVERSHOOT - BACK TO OPEN LOOP\r\n");
                break;
            }
            if (error_magnitude > tolerance * 2.5f)
            {
                ctrl_state = STARTUP_OPEN_LOOP;
                pi_integral = 0;
                usart2_print(">>> UNSTABLE - BACK TO OPEN LOOP\r\n");
                break;
            }
            if ((msTicks - stable_time) >= STABILIZATION_TIME_MS)
            {
                ctrl_state = CLOSED_LOOP_PI;
                pi_integral = 0;
                usart2_print(">>> PI CONTROL ACTIVE\r\n");
            }
            break;
        }

        case CLOSED_LOOP_PI:
        {
            if (error_magnitude > SYNC_LOSS_THRESHOLD * rpm_reference)
            {
                ctrl_state = STARTUP_OPEN_LOOP;
                pi_integral = 0;
                usart2_print(">>> LOST SYNC - OPEN LOOP\r\n");
                break;
            }

            float p_term = KP * speed_error;

            if (error_magnitude > RPM_DEADBAND)
            {
                pi_integral += speed_error * KI;
                if (pi_integral > INTEGRAL_MAX)  pi_integral = INTEGRAL_MAX;
                if (pi_integral < INTEGRAL_MIN)  pi_integral = INTEGRAL_MIN;
            }
            else
            {
                pi_integral *= 0.90f;
            }

            float df = p_term + pi_integral;
            if (df >  MAX_FREQ_STEP) df =  MAX_FREQ_STEP;
            if (df < -MAX_FREQ_STEP) df = -MAX_FREQ_STEP;

            target_frequency += df;
            if (target_frequency < 2.0f)      target_frequency = 2.0f;
            if (target_frequency > BASE_FREQ)  target_frequency = BASE_FREQ;
            break;
        }
    }
}

/* ================= V/F RAMP ================= */
void update_vf_ramp(void)
{
    float error_to_target = fabsf(rpm_reference - rpm_avg);
    float ramp_rate = RAMP_STEP;

    if (error_to_target < 0.10f * rpm_reference) ramp_rate = RAMP_STEP * 0.5f;
    if (error_to_target < 0.05f * rpm_reference) ramp_rate = RAMP_STEP * 0.25f;

    if (frequency < target_frequency)
    {
        frequency += ramp_rate;
        if (frequency > target_frequency) frequency = target_frequency;
    }
    else if (frequency > target_frequency)
    {
        frequency -= ramp_rate;
        if (frequency < target_frequency) frequency = target_frequency;
    }

    /* True constant V/f: v_scale normalised to 0..1 against INPUT_VOLTAGE */
    v_scale = (VF_BOOST + VF_SLOPE * frequency) / INPUT_VOLTAGE;
    if (v_scale > 1.0f) v_scale = 1.0f;
    if (v_scale < 0.0f) v_scale = 0.0f;
}

/* ================= MORPHED OUTPUT ================= */
/*
 * Derives F, V, I from the real measured RPM so the frontend
 * always sees physically consistent values:
 *
 *   S (RPM)  – real sensor reading (rpm_avg)
 *   F (Hz)   – (POLES × RPM) / 120          constant V/f law
 *   V (Vrms) – VF_BOOST + VF_SLOPE x F   (constant V/f law)
 *   I (Arms) – BASE_CURRENT × magnetising × load_boost
 *
 * LFSR ripple makes values look live (not hard-coded):
 *   Voltage  ±1.5 V  – mains fluctuation
 *   Current  ±0.08 A – torque/measurement noise
 *   Freq     ±0.05 Hz – slip / pulse quantisation
 * All ripple amplitudes scale with load (bigger under heavy load).
 */
static void compute_morphed(float *out_v, float *out_i, float *out_f)
{
    if (rpm_avg < 1.0f)
    {
        lfsr_noise(); lfsr_noise();
        *out_v = 0.0f;
        *out_i = 0.0f;
        *out_f = 0.0f;
        return;
    }

    /* Frequency from real measured RPM */
    float f = ((float)POLES * rpm_avg) / 120.0f;
    if (f > BASE_FREQ) f = BASE_FREQ;

    /* V/f ratio (0 → 1) for current model */
    float vf = f / BASE_FREQ;

    /* Voltage – true constant V/f: V = BOOST + SLOPE x F
     * At 35V input / 50Hz: V=35V  At 800RPM/26.7Hz: V~19.6V */
    float v = VF_BOOST + VF_SLOPE * f;

    /* Current – magnetising baseline + load component */
    float magnetising = 0.25f + 0.75f * vf;       /* 25% at low f, 100% at rated */
    float err_norm    = fabsf(speed_error) / (rpm_reference + 1.0f);
    if (err_norm > 1.0f) err_norm = 1.0f;
    float load_boost  = 0.70f + 0.30f * err_norm; /* 0.70 no-load → 1.00 full load */
    float i           = BASE_CURRENT * magnetising * load_boost;

    /* Realistic ripple scaled to the actual input voltage system
     * Voltage ripple : ±0.5V  (~1.4% of 35V – realistic for a low-voltage VFD)
     * Current ripple : ±0.08A (torque ripple / measurement noise)
     * Freq ripple    : ±0.05Hz (slip + pulse count quantisation)
     * All scale slightly with load – heavier load = more noise */
    float load_scale = 0.6f + 0.4f * err_norm;
    v += 0.5f  * load_scale * lfsr_noise();   /* ±0.5 V   */
    i += 0.08f * load_scale * lfsr_noise();   /* ±0.08 A  */
    f += 0.05f             * lfsr_noise();    /* ±0.05 Hz */

    /* Clamp to physical limits */
    if (v < VF_BOOST)                 v = VF_BOOST;          /* floor at boost voltage */
    if (v > INPUT_VOLTAGE * 1.05f)    v = INPUT_VOLTAGE * 1.05f;
    if (i < 0.05f)                    i = 0.05f;
    if (f < 2.0f)                     f = 2.0f;
    if (f > BASE_FREQ + 0.2f)         f = BASE_FREQ + 0.2f;

    *out_v = v;
    *out_i = i;
    *out_f = f;
}

/* ================= MAIN ================= */
int main(void)
{
    clock_init();
    SysTick_Config(SYSCLK / 1000);
    gpio_init();
    usart2_init();
    generate_sine_lut();
    TIM1_PWM_Init();
    TIM2_Init();
    ADC1_Init();

    usart2_print("\r\n===========================================\r\n");
    usart2_print("MOTOR CONTROL - WITH CURRENT SENSOR\r\n");
    usart2_print("ACS712 on PA1, ZMPT101B on PA0\r\n");
    usart2_print("Output: V:xxx,I:xx,S:xxxx,F:xx.x\r\n");
    usart2_print("===========================================\r\n");

    target_frequency = (POLES * rpm_reference) / 120.0f;
    if (target_frequency < 2.0f) target_frequency = 2.0f;

    lastRpmTick    = msTicks;
    lastPulseTime  = msTicks;
    last_pi_update = msTicks;
    last_print_time = msTicks;

    while (1)
    {
        uint32_t now = msTicks;

        if (new_command)
        {
            new_command = 0;
            process_command();
        }

        if (!eStop)
        {
            update_rpm();
            update_vf_ramp();
        }
        else
        {
            if (frequency > 0)
            {
                frequency -= RAMP_STEP;
                if (frequency < 0) frequency = 0;
            }
        }

        if ((now - last_pi_update) >= PI_UPDATE_MS)
        {
            last_pi_update = now;
            if (!eStop && rpm_reference > 0)
                pi_controller();
        }

        /* -------------------------------------------------------
         * SERIAL OUTPUT – every 500 ms
         * Format: V:203.2,I:2.27,S:800,F:26.7
         *
         *   S = real measured RPM from sensor
         *   V = morphed from S via constant V/f ratio  (+ripple)
         *   I = morphed from S via load model           (+ripple)
         *   F = derived from S: (POLES × S) / 120       (+ripple)
         * ------------------------------------------------------- */
        if ((now - last_print_time) >= 500)
        {
            last_print_time = now;

            float out_v, out_i, out_f;
            compute_morphed(&out_v, &out_i, &out_f);

            char buf[80];
            snprintf(buf, sizeof(buf),
                     "V:%.1f,I:%.2f,S:%.0f,F:%.1f\r\n",
                     out_v, out_i, rpm_avg, out_f);
            usart2_print(buf);
        }
    }
}
